var y: int
var iters: int
func main(): int {
    y = 5
    var total: int

    total = sum()
    return iters + total*10
}

func sum(): int {
    var total: int
    total=0
    iters=0
    while(true){
        total = total + y
        y = y-1
        if( y == 0 ){
            return total
        }
        iters = iters+1
    }
    return 0    //prevent errors
}
